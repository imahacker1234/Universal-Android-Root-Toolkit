#!/data/data/com.termux/files/usr/bin/bash

# ╔══════════════════════════════════════════════════════════════════╗
# ║     Universal Android Root & Recovery Toolkit v3.0               ║
# ║          One-Line Installer - curl | bash                        ║
# ╚══════════════════════════════════════════════════════════════════╝

set -e

REPO_URL="https://github.com/YOUR_USERNAME/Universal-Android-Root-Toolkit.git"
RAW_URL="https://raw.githubusercontent.com/YOUR_USERNAME/Universal-Android-Root-Toolkit/main"
INSTALL_DIR="$HOME/Universal-Android-Root-Toolkit"
LOG_FILE="$HOME/uroot_install.log"

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC='\033[0m'

# Logging
exec 1> >(tee -a "$LOG_FILE")
exec 2>&1

echo -e "${CYAN}"
echo "╔══════════════════════════════════════════════════════════════════╗"
echo "║                                                                  ║"
echo "║   ██╗   ██╗███╗   ██╗██╗██╗   ██╗███████╗██████╗ ███████╗        ║"
echo "║   ██║   ██║████╗  ██║██║██║   ██║██╔════╝██╔══██╗██╔════╝        ║"
echo "║   ██║   ██║██╔██╗ ██║██║██║   ██║█████╗  ██████╔╝███████╗        ║"
echo "║   ██║   ██║██║╚██╗██║██║╚██╗ ██╔╝██╔══╝  ██╔══██╗╚════██║        ║"
echo "║   ╚██████╔╝██║ ╚████║██║ ╚████╔╝ ███████╗██║  ██║███████║        ║"
echo "║    ╚═════╝ ╚═╝  ╚═══╝╚═╝  ╚═══╝  ╚══════╝╚═╝  ╚═╝╚══════╝        ║"
echo "║                                                                  ║"
echo "║          Universal Android Root & Recovery Toolkit               ║"
echo "║                      Version 3.0                                 ║"
echo "╚══════════════════════════════════════════════════════════════════╝"
echo -e "${NC}"

# Check if running in Termux
if [ -z "$PREFIX" ] || [ ! -d "$PREFIX" ]; then
    echo -e "${RED}[✗] Error: This installer must run in Termux!${NC}"
    echo -e "${YELLOW}[!] Install Termux from F-Droid first${NC}"
    exit 1
fi

echo -e "${CYAN}[*] Starting installation...${NC}"
echo -e "${CYAN}[*] Log file: $LOG_FILE${NC}\n"

# Step 1: Update packages
echo -e "${YELLOW}[1/7] Updating packages...${NC}"
yes | pkg update -y && yes | pkg upgrade -y

# Step 2: Install dependencies
echo -e "${YELLOW}[2/7] Installing dependencies...${NC}"
yes | pkg install -y git wget curl unzip python python-pip bc termux-api libffi openssl

# Step 3: Setup storage
echo -e "${YELLOW}[3/7] Setting up storage...${NC}"
termux-setup-storage

# Step 4: Remove old installation
if [ -d "$INSTALL_DIR" ]; then
    echo -e "${YELLOW}[4/7] Removing old installation...${NC}"
    rm -rf "$INSTALL_DIR"
fi

# Step 5: Clone repository
echo -e "${YELLOW}[5/7] Downloading toolkit...${NC}"
if command -v git &>/dev/null; then
    git clone --depth 1 "$REPO_URL" "$INSTALL_DIR"
else
    # Fallback: download as zip
    mkdir -p "$INSTALL_DIR"
    wget -q "https://github.com/YOUR_USERNAME/Universal-Android-Root-Toolkit/archive/refs/heads/main.zip" -O "$HOME/uroot.zip"
    unzip -q "$HOME/uroot.zip" -d "$HOME/"
    mv "$HOME/Universal-Android-Root-Toolkit-main"/* "$INSTALL_DIR/"
    rm -rf "$HOME/uroot.zip" "$HOME/Universal-Android-Root-Toolkit-main"
fi

# Step 6: Setup permissions
echo -e "${YELLOW}[6/7] Setting up permissions...${NC}"
cd "$INSTALL_DIR"
chmod -R +x *.sh modules/*.sh lib/*.sh 2>/dev/null || true

# Step 7: Install ADB/Fastboot
echo -e "${YELLOW}[7/7] Installing ADB & Fastboot...${NC}"
yes | pkg uninstall termux-adb 2>/dev/null || true

# Install termux-adb
curl -s https://raw.githubusercontent.com/nohajc/termux-adb/master/install.sh | bash

# Create symlinks
ln -sf "$PREFIX/bin/termux-adb" "$PREFIX/bin/adb" 2>/dev/null || true
ln -sf "$PREFIX/bin/termux-fastboot" "$PREFIX/bin/fastboot" 2>/dev/null || true

# Create global launcher command
echo -e "${CYAN}[*] Creating launcher...${NC}"
cat > "$PREFIX/bin/uroot" << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
cd "$HOME/Universal-Android-Root-Toolkit" && ./start.sh "$@"
EOF
chmod +x "$PREFIX/bin/uroot"

# Create Termux shortcut
mkdir -p "$HOME/.shortcuts"
cat > "$HOME/.shortcuts/Root-Toolkit" << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
cd "$HOME/Universal-Android-Root-Toolkit" && ./start.sh "$@"
EOF
chmod +x "$HOME/.shortcuts/Root-Toolkit"

# Create quick commands
cat > "$PREFIX/bin/uroot-flash" << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
cd "$HOME/Universal-Android-Root-Toolkit" && ./modules/advanced-flash.sh "$@"
EOF
chmod +x "$PREFIX/bin/uroot-flash"

cat > "$PREFIX/bin/uroot-root" << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
cd "$HOME/Universal-Android-Root-Toolkit" && ./modules/universal-root.sh "$@"
EOF
chmod +x "$PREFIX/bin/uroot-root"

# Installation complete
echo -e "\n${GREEN}"
echo "╔══════════════════════════════════════════════════════════════════╗"
echo "║                    ✓ INSTALLATION COMPLETE!                      ║"
echo "╠══════════════════════════════════════════════════════════════════╣"
echo "║                                                                  ║"
echo "║   Launch the toolkit with:                                      ║"
echo "║       uroot                                                       ║"
echo "║                                                                  ║"
echo "║   Quick commands:                                                 ║"
echo "║       uroot-root    - Direct to root installer                   ║"
echo "║       uroot-flash   - Direct to flash tool                       ║"
echo "║                                                                  ║"
echo "║   Or run manually:                                                ║"
echo "║       cd ~/Universal-Android-Root-Toolkit                         ║"
echo "║       ./start.sh                                                  ║"
echo "║                                                                  ║"
echo "╚══════════════════════════════════════════════════════════════════╝"
echo -e "${NC}"

echo -e "${CYAN}Next Steps:${NC}"
echo "  1. Connect your Android device via USB"
echo "  2. Enable USB Debugging: Settings → Developer Options"
echo "  3. Run 'uroot' to start the toolkit"
echo "  4. Grant Termux storage permission if prompted"
echo ""

# Cleanup log
rm -f "$LOG_FILE"

read -p "$(echo -e ${CYAN}[?]${NC} Launch toolkit now? [Y/n]: )" launch
if [[ ! "$launch" =~ [Nn] ]]; then
    uroot
fi
