# Check dependencies
check_environment() {
    local deps=("git" "wget" "curl" "unzip" "python" "bc")
    local missing=()
    
    for dep in "${deps[@]}"; do
        if ! command -v "$dep" &>/dev/null; then
            missing+=("$dep")
        fi
    done
    
    if [ ${#missing[@]} -gt 0 ]; then
        echo -e "${YELLOW}[!] Installing missing packages...${NC}"
        yes | pkg install "${missing[@]}" termux-api -y 2>/dev/null || true
    fi
    
    # Ensure ADB/Fastboot links exist
    [ ! -f "$PREFIX/bin/adb" ] && ln -sf "$PREFIX/bin/termux-adb" "$PREFIX/bin/adb" 2>/dev/null || true
    [ ! -f "$PREFIX/bin/fastboot" ] && ln -sf "$PREFIX/bin/termux-fastboot" "$PREFIX/bin/fastboot" 2>/dev/null || true
}

# Device status panel
show_device_panel() {
    echo -e "\n${CYAN}╔══════════════════════════════════════════════════════════════════╗${NC}"
    echo -e "${CYAN}║${WHITE}                    DEVICE STATUS                                  ${CYAN}║${NC}"
    echo -e "${CYAN}╠══════════════════════════════════════════════════════════════════╣${NC}"
    
    # ADB Status
    local adb_device=$(adb devices 2>/dev/null | grep -v "List" | grep "device" | awk '{print $1}')
    if [ -n "$adb_device" ]; then
        local model=$(adb shell getprop ro.product.model 2>/dev/null | tr -d '\r' || echo "Unknown")
        local android=$(adb shell getprop ro.build.version.release 2>/dev/null | tr -d '\r' || echo "?")
        echo -e "${CYAN}║${GREEN}  ✓ ADB Connected:${NC} $model (Android $android)${CYAN}$(printf '%*s' $((49-${#model}-${#android})) '')║${NC}"
    else
        echo -e "${CYAN}║${YELLOW}  ! No ADB Connection${NC}$(printf '%*s' 46 '')${CYAN}║${NC}"
    fi
    
    # Fastboot Status
    local fb_device=$(fastboot devices 2>/dev/null | grep "fastboot" | awk '{print $1}')
    if [ -n "$fb_device" ]; then
        echo -e "${CYAN}║${GREEN}  ✓ Fastboot Connected:${NC} $fb_device${CYAN}$(printf '%*s' $((43-${#fb_device})) '')║${NC}"
    else
        echo -e "${CYAN}║${YELLOW}  ! No Fastboot Connection${NC}$(printf '%*s' 42 '')${CYAN}║${NC}"
    fi
    
    # Root Status
    if adb shell su -c "id" &>/dev/null; then
        echo -e "${CYAN}║${GREEN}  ✓ Root Access Available${NC}$(printf '%*s' 42 '')${CYAN}║${NC}"
    else
        echo -e "${CYAN}║${RED}  ✗ No Root Access${NC}$(printf '%*s' 47 '')${CYAN}║${NC}"
    fi
    
    echo -e "${CYAN}╚══════════════════════════════════════════════════════════════════╝${NC}\n"
}
